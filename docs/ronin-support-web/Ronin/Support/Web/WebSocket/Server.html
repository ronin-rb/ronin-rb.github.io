def initialize(url, bind_host: nil,
                    bind_port: nil,
                    backlog:   5,
                    ssl:       {})
  super(url)

  @bind_host = bind_host
  @bind_port = bind_port || @port

  @socket = case @url.scheme
            when 'ws'
              Support::Network::TCP.server(
                host:    @bind_host,
                port:    @bind_port,
                backlog: backlog
              )
            when 'wss'
              Support::Network::SSL.server(
                host:    @bind_host,
                port:    @bind_port,
                backlog: backlog,
                **ssl
              )
            else
              raise(ArgumentError,"unsupported websocket scheme: #{url}")
            end
end
