def initialize(name,operands)
  operands = operands.map do |value|
    case value
    when Integer, nil then ImmediateOperand.new(value)
    else                   value
    end
  end

  super(name,operands)
end

def width
  self.operands.map { |op|
    op.width if op.respond_to?(:width)
  }.compact.max
end
