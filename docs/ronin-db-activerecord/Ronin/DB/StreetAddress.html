def self.for_organization(name)
  joins(organization_street_addresses: :organization).where(
    organization_street_addresses: {
      ronin_organizations: {name: name}
    }
  )
end

def to_s
  string = "#{address}#{$/}#{city}"
  string << ", #{state}"  if state
  string << " #{zipcode}" if zipcode
  string << "#{$/}#{country}"
  return string
end
