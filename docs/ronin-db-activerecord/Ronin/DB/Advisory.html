def url
  case prefix
  when 'CVE'  then "https://nvd.nist.gov/vuln/detail/#{id}"
  when 'RHSA' then "https://access.redhat.com/errata/#{id}"
  when 'GHSA' then "https://github.com/advisories/#{id}"
  end
end
