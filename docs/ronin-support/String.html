def encode_bytes(include: nil, exclude: nil)
  included  = (Chars::CharSet.new(*include) if include)
  excluded  = (Chars::CharSet.new(*exclude) if exclude)
  formatted = String.new(encoding: ::Encoding::UTF_8)

                   yield(b)
                 else
                   b
                 end
  end
