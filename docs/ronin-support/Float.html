Packs the Float into a String.

def pack(argument, **kwargs)
  case argument
  when String
    [self].pack(argument)
  else
    types = Ronin::Support::Binary::CTypes.platform(**kwargs)
    type  = types[argument]
    type.pack(self)
  end
end
