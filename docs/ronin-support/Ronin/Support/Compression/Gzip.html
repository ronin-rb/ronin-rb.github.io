def self.new(io, mode: 'r', &block)
  gzip_class = if mode.include?('w') || mode.include?('a')
                 Writer
               elsif mode.include?('r')
                 Reader
               else
                 raise(ArgumentError,"mode argument must include either 'r', 'w', or 'a': #{mode.inspect}")
               end

  return gzip_class.wrap(io,&block)
end
