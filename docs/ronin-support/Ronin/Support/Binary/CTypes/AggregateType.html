def pack(value)
  if @pack_string
    values = []
    enqueue_value(values,value)

    return values.pack(@pack_string)
  else
    raise(NotImplementedError,"#{self.class} does not define a #pack_string")
  end
end

def unpack(data)
  if @pack_string
    values = data.unpack(@pack_string)

    return dequeue_value(values)
  else
    raise(NotImplementedError,"#{self.class} does not define a #pack_string")
  end
end
