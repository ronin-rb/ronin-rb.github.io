def initialize_type_system(type_system: nil,
                           endian:      nil,
                           arch:        nil,
                           os:          nil)
  @endian = endian
  @arch   = arch
  @os     = os

  @type_system = type_system || CTypes.platform(
                                  endian: endian,
                                  arch:   arch,
                                  os:     os
                                )

  @type_resolver = CTypes::TypeResolver.new(@type_system)
end
