def enqueue_value(values,array)
  case array
  when Binary::Array
    values.push(array.to_s)
  when ::Array
    values.push(@array_type.pack(array))
  end
end

def pack(array)
  case array
  when Binary::Array
    array.to_s
  when ::Array
    @array_type.pack(array)
  else
    raise(ArgumentError,"value must be either a #{Binary::Array} or an #{::Array}: #{array.inspect}")
  end
end
