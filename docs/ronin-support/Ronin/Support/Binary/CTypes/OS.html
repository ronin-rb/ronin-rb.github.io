def typedef(type,new_name)
  case type
  when Type   then @typedefs[new_name] = type
  when Symbol then @typedefs[new_name] = self[type]
  else
    raise(ArgumentError,"type must be either a Symbol or a #{Type}: #{type.inspect}")
  end
end
