def enqueue_value(values,struct)
  case struct
  when Binary::Struct
    values.push(struct.to_s)
  when Hash
    values.push(@struct_type.pack(struct))
  end
end

def pack(struct)
  case struct
  when Binary::Struct
    struct.to_s
  when Hash
    @struct_type.pack(struct)
  else
    raise(ArgumentError,"value must be either a #{Binary::Struct} or an #{Hash}: #{struct.inspect}")
  end
end
