def pack(value)
  if @pack_string
    [value].pack(@pack_string)
  else
    raise(NotImplementedError,"#{self.class} does not define a #pack_string")
  end
end

def unpack(data)
  if @pack_string
    data.unpack1(@pack_string)
  else
    raise(NotImplementedError,"#{self.class} does not define a #pack_string")
  end
end
