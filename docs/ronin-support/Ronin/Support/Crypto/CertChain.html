def self.parse(string)
  cert_buffer = String.new
  certs = []

  string.each_line do |line|
    cert_buffer << line

    if line.chomp == '-----END CERTIFICATE-----'
      certs << Cert.parse(cert_buffer)
      cert_buffer.clear
    end
  end

  return new(certs)
end
