def self.new(io_or_buffer, mode: 'r', &block)
  io = case io_or_buffer
       when String then StringIO.new(io_or_buffer,mode)
       else             io_or_buffer
       end

  return super(io,&block)
end

def self.open(path,&block)
  if block
    File.open(path,'rb') do |file|
      new(file,&block)
    end
  else
    new(File.new(path,'rb'))
  end
end
