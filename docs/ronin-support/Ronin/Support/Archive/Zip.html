def self.new(path, mode: 'r', &block)
  zip_class = if mode.include?('w') || mode.include?('a')
                Writer
              elsif mode.include?('r')
                Reader
              else
                raise(ArgumentError,"mode argument must include either 'r', 'w', or 'a': #{mode.inspect}")
              end

  return zip_class.new(path,&block)
end
