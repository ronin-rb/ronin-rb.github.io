def self.calculate(string, base: 2)
  char_counts = Hash.new(0)

  string.each_char do |char|
    char_counts[char] += 1
  end

  length  = string.length.to_f
  entropy = 0.0

  char_counts.each_value do |count|
    freq     = count / length
    entropy -= freq * Math.log(freq,base)
  end

  return entropy
end
