def self.table(char_set=nil)
  if char_set
    CHAR_SETS.fetch(char_set) do
      raise(ArgumentError,"unknown homoglyph character set (#{char_set.inspect}), must be #{CHAR_SETS.keys.map(&:inspect).join(', ')}")
    end
  else
    DEFAULT
  end
end
