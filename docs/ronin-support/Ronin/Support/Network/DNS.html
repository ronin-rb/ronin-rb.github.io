def self.resolver(nameservers: nil, nameserver: nil)
  if nameserver
    Resolver.new(nameserver: nameserver.to_s)
  elsif nameservers
    Resolver.new(nameservers: nameservers.map(&:to_s))
  else
    @resolver ||= Resolver.new(nameservers: self.nameservers)
  end
end
