def esmtp_connect(host,**kwargs)
  if block_given?
    smtp_connect(host,**kwargs) do |smtp|
      smtp.esmtp = true
      yield smtp
    end
  else
    smtp = smtp_connect(host,**kwargs)

    smtp.esmtp = true
    return smtp
  end
end
