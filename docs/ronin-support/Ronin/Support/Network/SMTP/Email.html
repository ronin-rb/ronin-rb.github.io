def initialize(from:       nil,
               to:         nil,
               subject:    nil,
               date:       Time.now,
               message_id: nil,
               headers:    nil,
               body:       nil)
  @from       = from
  @to         = to
  @subject    = subject
  @date       = date
  @message_id = message_id

  @headers = {}
  @headers.merge!(headers) if headers

  @body = []

  if body
    case body
    when Array
      @body += body
    else
      @body << body
    end
  end

  yield self if block_given?
end

def to_s
  message = []

  if @from
    message << "From: #{@from}"
  end

  if @to
    message << case @to
               when Array
                 "To: #{@to.join(', ')}"
               else
                 "To: #{@to}"
               end
  end

  if @subject
    message << "Subject: #{@subject}"
  end

  if @date
    message << "Date: #{@date}"
  end

  if @message_id
    message << "Message-Id: <#{@message_id}>"
  end

  @headers.each do |name,value|
    message << "#{name}: #{value}"
  end

  message << ''
  message += @body

  return message.join(CRLF)
end
