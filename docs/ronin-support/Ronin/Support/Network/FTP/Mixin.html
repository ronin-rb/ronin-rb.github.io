def ftp_connect(host, port:     DEFAULT_PORT,
                      user:     DEFAULT_USER,
                      password: nil,
                      account:  nil,
                      passive:  true)
  host = DNS::IDN.to_ascii(host)
  ftp  = Net::FTP.new

  ftp.connect(host,port)
  ftp.login(user,password,account)
  ftp.passive = passive

  if block_given?
    yield ftp
    ftp.close
  else
    return ftp
  end
end
