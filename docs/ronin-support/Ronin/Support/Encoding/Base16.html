def self.decode(data)
  decoded = String.new(encoding: Encoding::ASCII_8BIT)

  data.scan(/../).each do |hex_char|
    decoded << hex_char.to_i(16).chr
  end

  return decoded
end

def self.encode(data)
  encoded = String.new

  data.each_byte do |byte|
    encoded << ("%.2x" % byte)
  end

  return encoded
end
