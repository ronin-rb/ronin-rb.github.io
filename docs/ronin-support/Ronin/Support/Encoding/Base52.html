def self.decode(string)
  decoded    = 0
  multiplier = 1
  table_size = TABLE.length

  string.each_char.reverse_each.each_with_index do |char,index|
    decoded    += TABLE.index(char) * multiplier
    multiplier *= table_size
  end

  return decoded
end

def self.encode_int(int)
  if int == 0
    String.new('0')
  elsif int < 0
    raise(ArgumentError,"cannot Base52 encode negative numbers: #{int.inspect}")
  else
    encoded    = String.new
    table_size = TABLE.length

    while int > 0
      encoded.prepend(TABLE[int % table_size])
      int /= table_size
    end

    return encoded
  end
end
