def initialize(name, method_name: name,
                     usage:       nil,
                     completions: nil,
                     summary:     ,
                     help:        summary)
  @name        = name
  @method_name = method_name
  @usage       = usage
  @summary     = summary
  @help        = help
  @completions = completions
end

def to_s
  if @usage
    "#{@name} #{@usage}"
  else
    @name.to_s
  end
end
