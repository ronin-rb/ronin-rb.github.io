def self.included(command)
  command.option :param, short: '-p',
                         value: {
                           type:  /\A[^=]+=.+\z/,
                           usage: 'NAME=VALUE'
                         },
                         desc: 'Sets a param' do |str|
                           name, value = str.split('=',2)

                           @params[name.to_sym] = value
                         end
end
