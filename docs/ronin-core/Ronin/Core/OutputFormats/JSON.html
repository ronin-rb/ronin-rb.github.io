def <<(value)
  unless value.respond_to?(:to_json)
    raise(NotImplementedError,"output value does not define a #to_json method: #{value.inspect}")
  end

  if @first_value
    @io.write("[#{value.to_json}")
    @first_value = false
  else
    @io.write(",#{value.to_json}")
  end

  @io.flush
  return self
end

def close

  else
    @io.write("]")
  end

  super
