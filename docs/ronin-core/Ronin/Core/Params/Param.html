def coerce(value)
  case value
  when nil
    if required?
      raise(ValidationError,"param requires a non-nil value")
    end
  else
    @type.coerce(value)
  end
end

def default_value
  if @default.respond_to?(:call) then @default.call
  else                                @default.dup
  end
end
