def self.extract(response)
  CLASSES.each do |klass|
    if (session_cookie = klass.extract(response))
      return session_cookie
    end
  end

  return nil
end

def self.parse(string)
  CLASSES.each do |klass|
    if klass.identify?(string)
      return klass.parse(string)
    end
  end

  return nil
end
