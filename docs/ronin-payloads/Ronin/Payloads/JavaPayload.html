def compile(*source_files, dest_dir: nil)
  args = [params[:javac]]
  args << '-d' << dest_dir if dest_dir
  args.concat(source_files)

  case system(*args)
  when false
    raise(BuildFailed,"javac command failed: #{args.join(' ')}")
  when nil
    raise(BuildFailed,"javac command not installed")
  end
end
