def build
  @java_file  = File.join(build_dir,'Payload.java')
  @class_file = File.join(build_dir,'Payload.class')

  File.write(@java_file,erb(TEMPLATE))
  compile(@java_file, dest_dir: build_dir)

  @payload = File.binread(@class_file)
end
