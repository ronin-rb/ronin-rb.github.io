def compile_ts(*source_files)
  args = [params[:tsc]]
  args.concat(source_files)

  case system(*args)
  when false
    raise(BuildFailed,"tsc command failed: #{args.join(' ')}")
  when nil
    raise(BuildFailed,"tsc command not installed")
  end
end
