def pack_ipv4(ip, negate: false)
  ipaddr = IPAddr.new(ip)

  unless ipaddr.ipv4?
    raise(ArgumentError,"IP must be an IPv4 address: #{ip.inspect}")
  end

  ipaddr = ~ipaddr if negate
  ipaddr.hton
end
