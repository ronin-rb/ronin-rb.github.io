def packed_ipv6(**kwargs)
  unless (ip = host_ipv6_address)
    unless (ip = host_ipv4_address)
      raise(BuildFailed,"host name has no IP addresses: #{params[:host].inspect}")
    end

    ip = "::ffff:#{ip}"
  end

  pack_ipv6(ip,**kwargs)
end
