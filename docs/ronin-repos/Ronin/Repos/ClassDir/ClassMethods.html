def list_files
  modules = Set.new(super)
  pattern = File.join(repo_class_dir,"{**/}*.rb") 

  Repos.list_files(pattern).each do |path|
    modules << path.chomp('.rb')
  end

  return modules.to_a
end

def path_for(name)

    Repos.find_file(File.join(repo_class_dir,"#{name}.rb"))
end
