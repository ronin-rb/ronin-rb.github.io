def headers
  headers = {}

  env.each do |name,value|
    if name =~ /^HTTP_/
      header_words = name[5..].split('_')
      header_words.each(&:capitalize!)

      header_name = header_words.join('-')

      headers[header_name] = value
    end
  end

  return headers
end

def ip_with_port
  if env.has_key?('REMOTE_PORT')
    "#{ip}:#{env['REMOTE_PORT']}"
  else
    ip
  end
