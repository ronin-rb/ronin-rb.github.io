def ssl=(ssl)
  if ssl
    self.port   = 443
    self.scheme = 'https'
  else
    self.port   = 80
    self.scheme = 'http'
  end

  return ssl
end

def xhr=(xhr)
  if xhr then @env['HTTP_X_REQUESTED_WITH'] = 'XMLHttpRequest'
  else        @env.delete('HTTP_X_REQUESTED_WITH')
  end

  return xhr
end
