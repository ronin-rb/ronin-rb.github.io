def process_value(service)
  host, port = host_and_port(service)

  begin
    banner = Support::Network::TCP.banner(host,port)

    if options[:with_host_port]
      puts "#{service}: #{banner}"
    else
      puts banner
    end
  rescue StandardError => error
    print_error("#{service}: #{error.message}")
  end
end
