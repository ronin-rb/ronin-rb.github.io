def process_input(input)
  input.each_line(chomp: true) do |line|
    line.scan(@pattern) do |match|
      puts match
    end
  end
end

def run(*files)
  unless @pattern
    print_error "must specify a pattern to search for"
    exit(-1)
  end

  super(*files)
end
